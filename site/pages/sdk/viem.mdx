# Viem

Extending viem with sushi actions.

```ts
import { createPublicClient, createWalletClient, http } from 'viem'
import { mainnet } from 'viem/chains'
import { publicActions, walletActions } from 'sushi'
 
const publicClient = createPublicClient({
  chain: mainnet,
  transport: http()
}).extends(publicActions)
const { request } = await client.simulateProcessRoute()
  .then(console.log)
  .catch(console.error)

const account = privateKeyToAccount('PRIVATE_KEY')
const walletClient = createWalletClient({
  account,
  chain: mainnet,
  transport: http()
}).extends(walletActions)
await client.processRoute(request)
  .then(console.log)
  .catch(console.error)

```
